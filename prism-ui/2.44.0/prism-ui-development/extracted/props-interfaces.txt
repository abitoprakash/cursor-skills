export interface DropdownProps extends Omit<PopupProviderProps, 'popupPlacement'> {
  /**
   * This is the component/text of the dropdown where the user can hover or click on. By default
   * if `text` is set then the `Button` type TEXT_NORMAL component will be used to wrap the text. For other component
   * types, beware of handling the `disabled` property. See `disabled` prop for details.
   */
  children: React.ReactNode;
  /**
   * Type of Dropdown. If not set, the default dropdown will be used.
   *
   * export enum DropdownType {<br />
   *   NAVIGATION = 'navigation'<br />
   * }
   */
  type?: DropdownType;
  /**
   * ButtonProps to be passed to the Button component when children is a string.
   */
  buttonProps?: ButtonProps;
  /**
   * Event to trigger PopupProvider.
   */
  action?: ('hover' | 'click' | 'focus' | 'contextMenu')[]
  | 'hover' | 'click' | 'focus' | 'contextMenu';
  /**
   * This disables the Dropdown component and will also pass down 'disabled' prop to child node.
   * Note: When used it is expected the child node knows how to respond to this
   * property accordingly.
   */
  disabled?: boolean;
  /**
   * Set offset position for popup. String values can be of any measurement type. e.g. px, vh, em<br />
   * type PopupOffSet = {<br />
   *   x?: string<br />
   *   y?: string<br />
   * };
   * The offset does not count the caret, to be predictable in its container position with or without the caret.
   */
  popupOffset?: PopupOffSet;
  /**
   * Set placement position of dropdown over children.
   */
  popupPlacement?: 'auto' | PopupProviderProps['popupPlacement'];
  /**
   * Wrap `popup` inside the styled dropdown container. Note: When set to false, the styled
   * dropdown container will not be used and the dropdown automatic event handling for popup
   * visibility will cease function.
   */
  wrapComponent?: boolean;
  /**
   * Customize z-index of the dropdown container.
   * Default is 1610
   */
  zIndex?: number;
  /**
   * Background used for the dropdown container
   */
  background?: BackgroundTypeWithoutBody;
  /**
   * Flag used to disable default keyboard navigation
   */
  disableKeyboardNavigation?: boolean;
  /**
   * Id of the Dropdown component
   */
  id?: string;
  /**
   * Disable aria attributes on children.
   * Used when a child component handles aria attributes themselves.
   * Default is false
   */
  noAriaOnChildren?: boolean;
  /**
   * Focus first element on open
   * Note: This works only on dropdowns with wrapComponent set to true
   */
  focusOnOpen?: boolean;
}

export interface InfiniteScrollProps {
  /**
   * Array of items to be rendered.
   */
  data?: any[],
  /**
   * Type of element to render InfiniteScroll as.
   * (!) If you use a custom component ensure it has ref forwarding attached.
   *
   * If ReactNode is a function:
   *
   * @param {object} props - element props<br />
   * @param {number} [i] - index number of element<br />
   * @param {string} [type] - internal type of element
   * @returns React.ReactNode
   */
  listContainer?: React.ReactNode |
    ((props: Record<string, any>, i: number, type: string) => JSX.Element),
  /**
   * Type of element to render list item as.
   * (!) If you use a custom component ensure it has ref forwarding attached.
   *
   * if ReactNode is a function:
   *
   * @param {object} props - element props<br />
   * @param {number} index - index of element<br />
   * @param {string} type - 'data' or 'placeholder' type element
   * @returns React.ReactNode
   */
  listElement?: React.ReactNode |
    ((props: Record<string, any>, i: number, type: string) => JSX.Element),
  /**
   * Loader element
   */
  loader?: React.ReactNode,
  /**
   * Set initial scroll Y position.
   */
  initialScrollY?: {
    // Number of pixel to scroll down from the top
    scrollTop?: number,
    // The index from the props.data[] use scroll to. If this is used scrollTop will be ignore.
    index?: number
  },
  /**
   * Offset of data. Required when tracking dynamic data set.
   */
  offset?: number,
  /**
   * Handler for fetch request. Provides the following shape:
   *
   * @param {number} offset - offset of data<br />
   * @param {number} count - count of data
   * @returns {object}
   */
  onFetchRequest?: (value: { offset: number, count: number },
    context: Record<any, any>) => Record<any, any>,
  /**
   * Handler for scrolling. Provides the following shape:
   *
   * @param {event} event - scroll event<br />
   * @param {object} context - scroll context
   */
  onScroll?: (event: React.UIEvent, context: Record<any, any>) => void,
  /**
   *
   *Handler for update. Useful for sync of context e.g. pagination. Triggers on componentDidUpdate
   *and when the data range changes or when the scroll client changes in size.
   *
   * @param {object} context - scroll context
   */
  onUpdate?: (context: Record<any, any>) => void,
  /**
   * The page buffer size is used 1) when to call a fetch request and 2) provide a threshold for
   * optimized scrolling performance on data sets at extreme scale. This performance optimization
   * method is commonly known as virtual scrolling.
   *
   * A page buffer size of 2 means, 2 pages worth of rendered items will be prerended above the
   * viewport and 2 pages will be rendered below the viewport for a total of 5 pages at any given
   * time. If the current data set cannot sufficiently cater to the buffer needs then a fetch
   * request will be called for the optimal data set.
   *
   * Page size is automatically calculated by how many items can fit within the current viewport
   * even if the heights are dynamic.
   *
   * (!) It is recommended to increase the buffer size on smaller viewport, and decrease the
   * buffer size on smaller viewport for optimal performance.
   */

  pageBufferSize?: number,
  /**
   * Reference to a (parent) DOM element that is already providing the scroll container.
   *
   * (!) Mind that a parent element in React is mounted after the child. If passing an element via
   * a ref, a ref callback is generally required to sync the parent mount with the child mount.
   *
   * Pass a DOM element as return result from a function.
   * e.g. `() => document.querySelector('.container')`
   *
   * If the parent DOM is not available yet (e.g. still mounting), a sync callback is provided to
   * finalize the InfiniteScroll mount side effects. e.g. setting initial scroll position.
   *
   * @returns {Function} sync(element) - sync callback
   */
  scrollTarget?: HTMLElement | ((sync: (target: HTMLElement) => HTMLElement) => HTMLElement),
  /**
   * Scroll throttle in (ms). Larger number will mean slower call to scroll event listener.
   */
  scrollThrottle?: number,
  /**
   * Enable use of placeholders and track positions and total count of previous data.
   */
  scrollTracking: boolean,
  /**
   * Type of element to render spacer list item as. Spacer is used to maintain scrollbar size and
   * current position for virtual scrolling.
   * (!) If you use a custom component ensure it has ref forwarding attached.
   */
  spacerElement?: React.ReactNode,
  /**
   * Enable virtual scrolling. Recommended on to boost performance for large scale data sets.
   *
   * Note: When enabled, items outside of buffer threshold will be unmounted and remounted when
   * back inside buffer threshold. Any uncontrolled lifecycle will become restarted.
   *
   */
  virtualScrolling?: boolean
}
